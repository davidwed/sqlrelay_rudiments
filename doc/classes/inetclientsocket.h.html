#ifndef INETCLIENTSOCKET_H
#define INETCLIENTSOCKET_H

#include <rudiments/private/clienttransport.h>
#include <rudiments/private/inetsocket.h>

// The inetclientsocket class allows you to write programs that can talk to
// other programs across a network over TCP stream sockets.
//
// The inetclientsocket class provides methods for connecting to servers.
// Its ultimate parent class: transport provides methods for reading and 
// writing data and closing connections.

class inetclientsocket : public clienttransport, public inetsocket  {
        public:

                int     connectToServer(const char *hostname,
                                        unsigned short port,
                                        unsigned int retrytime,
                                        int tries);
                        // Convenience method that calls the
                        // initInetClientSocket() and connect()
                        // methods below.
                        //
                        // Returns 1 on success and 0 on failure.



                // These methods are split up in case you need to use them
                // seperately.  All three must be called in succession to
                // connect to a server.
                void    initInetClientSocket(const char *hostname,
                                                unsigned short port,
                                                unsigned int retrytime,
                                                int tries);
                        // Initializes the class to use "hostname", "port",
                        // "retrytime" and "tries" when connect() is called.
                int     connect();
                        // Attempts to connect to the "hostname" and "port" set
                        // earlier.  If the connection fails, it will retry
                        // "tries" times, waiting "retrytime" seconds between
                        // tries.  If "hostname" resolves to multiple
                        // addresses, each address will be tried "tries" times.
                        //
                        // Setting "tries" to 0 will cause it to try to 
                        // connect indefinitely.  Setting "retrytime" to 0
                        // will cause it to try to connect over and over
                        // as fast as possible (not recommended).
                        //
                        // Returns 1 on success and 0 on failure.
};

#include <rudiments/private/inetclientsocketinlines.h>

#endif
