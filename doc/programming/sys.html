Introduction

The sys class provides methods for getting information about the operating system and controlling the state of the operating system.


Getting System Information

In this example, various bits of system information are printed out.

#include <rudiments/sys.h>
#include <rudiments/charstring.h>
#include <rudiments/stdio.h>

int main(int argc, const char **argv) {

        char    *osname=sys::getOperatingSystemName();
        stdoutput.printf("OS Name               : %s\n",osname);
        delete[] osname;

        char    *release=sys::getOperatingSystemRelease();
        stdoutput.printf("OS Release    : %s\n",release);
        delete[] release;

        char    *version=sys::getOperatingSystemVersion();
        stdoutput.printf("OS Version    : %s\n",version);
        delete[] version;

        char    *arch=sys::getOperatingSystemArchitecture();
        stdoutput.printf("OS Arch               : %s\n",arch);
        delete[] arch;

        char    *hostname=sys::getHostName();
        stdoutput.printf("Host Name     : %s\n",hostname);
        delete[] hostname;

        double  onemin;
        double  fivemin;
        double  fifteenmin;
        sys::getLoadAverages(&onemin,&fivemin,&fifteenmin);
        stdoutput.printf("Load Averages : %0.2f  %0.2f  %0.2f\n",
                                        onemin,fivemin,fifteenmin);

        stdoutput.printf("Max Cmd Line Arg Length"
                        "                       : %lld\n",
                        sys::getMaxCommandLineArgumentLength());

        stdoutput.printf("Max Processes Per User"
                        "                       : %lld\n",
                        sys::getMaxProcessesPerUser());

        stdoutput.printf("Max Host Name Length"
                        "                       : %lld\n",

Note that not all systems support every piece of information. If a piece of information is unsupported, then a -1 is returned.


Reboot

This example reboots the system, when run as root or Administrator.

#include <rudiments/sys.h>
#include <rudiments/error.h>
#include <rudiments/stdio.h>

int main(int argc, const char **argv) {
        if (!sys::reboot()) {
                char    *err=error::getErrorString();
                stdoutput.printf("reboot failed: %s\n",err);
                delete[] err;
        }
}

Shutdown

This example performs a graceful system shutdown, when run as root or Administrator.

#include <rudiments/sys.h>
#include <rudiments/error.h>
#include <rudiments/stdio.h>

int main(int argc, const char **argv) {
        if (!sys::shutDown()) {
                char    *err=error::getErrorString();
                stdoutput.printf("shutdown failed: %s\n",err);
                delete[] err;
        }
}

Halt

This example performs a less-graceful system halt, when run as root or Administrator.

#include <rudiments/sys.h>
#include <rudiments/error.h>
#include <rudiments/stdio.h>

int main(int argc, const char **argv) {
        if (!sys::halt()) {
                char    *err=error::getErrorString();
                stdoutput.printf("halt failed: %s\n",err);
                delete[] err;
        }
}
