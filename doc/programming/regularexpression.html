Using the regularexpression class

The regularexpression class provides methods for making comparisons between text and regular expressions.

Regular expressions are complex, powerful, used in command line programs like grep, sed and find, and extensively in Perl.

#include <rudiments/regularexpression.h>
#include <rudiments/process.h>
#include <rudiments/stdio.h>

int main(int argc, const char **argv) {

        // use the static method for a quick yes or no match...
        const char      *string="void f(int a, bool b) { exit(0); }";
        const char      *pattern="(void|int|bool).*f\\(.*\\) { .* }";

        stdoutput.printf("quick match against \"%s\"...\n",pattern);
        if (regularexpression::match(string,pattern)) {
                stdoutput.printf("  \"%s\" matches\n\n",string);
        } else {
                stdoutput.printf("  \"%s\" doesn't match\n\n",string);
        }



        // compile expressions that need to be matched over and over...
        // (study complex patterns for better performance)
        regularexpression       re;
        re.compile(pattern);
        re.study();

        stdoutput.printf("compiled match against \"%s\"...\n",pattern);

        const char * const strings[]={
                "class t { public: int f(int a); void f(int a, bool b); };",
                "void f(int a, bool b) { exit(0); }",
                "int f(int a) { printf(\"hello\\n\"); }",
                "struct m { int a; int b; int c; };",
                NULL
        };
        for (const char * const *s=strings; *s; s++) {
                if (re.match(*s,pattern)) {
                        stdoutput.printf("  \"%s\" matches\n",*s);
                } else {
                        stdoutput.printf("  \"%s\" doesn't match\n",*s);
                }
        }
        stdoutput.write('\n');



        // some patterns produce multiple matches...
        string="hello there everyone";
        pattern="(\\w+) (\\w+)";

        re.compile(pattern);
        re.match(string);

        stdoutput.printf("\"%s\" matches \"%s\" %d times\n",
                                string,pattern,re.getSubstringCount());

        for (int32_t i=0; i<re.getSubstringCount(); i++) {
                stdoutput.printf("  match %d starts at: \"%s\"\n",
                                                i,re.getSubstringStart(i));
                stdoutput.printf("  match %d ends at  : \"%s\"\n",
                                                i,re.getSubstringEnd(i));
        }


        process::exit(0);
}
